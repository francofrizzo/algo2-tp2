\section{M\'{o}dulo Diccionario de Strings}

\Encabezado{Notas preliminares}
  En todos los casos, al indicar las complejidades de los algoritmos, las variables que se utilizan corresponden a:
  \vspace{-0.5em}\begin{itemize}
    \item $n$: Cantidad de claves definidas en el diccionario.
    \item $L$: Longitud de la clave m\'as larga definida en el diccionario.
  \end{itemize}

\begin{Interfaz}
  
  \begin{paramFormales}
    \paramGeneros{$\alpha$}

    \paramFuncion{
      \InterfazFuncion{$\bullet = \bullet$}{\In{a_{0}}{$\alpha$}, \In{a_{1}}{$\alpha$}}{bool}
      {$res \igobs$ equal($a_{1}$, $a_{2}$)}
      [$\Theta(equal(a_{1}, a_{2}))$]
      [funci\'on de igualdad de $\alpha$]
    }
    
    \paramFuncion{
      \InterfazFuncion{Copiar}{\In{a}{$\alpha$}}{$\alpha$}
      {$res \igobs a$}
      [$\Theta(copy(a))$]
      [funci\'{o}n de copia de $\alpha$]
    }

  \end{paramFormales}

  \seExplicaCon{Diccionario(Secuencia(Char), $\alpha$)}

  \generos{diccStr($\alpha$)}

  \Encabezado{Operaciones de diccionario}

    \InterfazFuncion{Vacio}{}{diccStr($\alpha$)}%
    {$res \igobs$ vac\'{i}o}%
    [$\Theta(1)$]%
    [Devuelve un diccionario vac\'io.]

    \InterfazFuncion{Definido}{\In{k}{string}, \In{d}{diccStr($\alpha$)}}{bool}%
    {$res \igobs$ def?($k$, $d$)}%
    [$\Theta(L)$]%
    [Devuelve \emph{true} si y solo si la clave $k$ est\'a definida en el diccionario.]

    \InterfazFuncion{Definir}{\In{k}{string}, \In{s}{$\alpha$}, \Inout{d}{diccStr($\alpha$)}}{}%
    [$d \igobs d_{0}$]%
    {$d \igobs$ definir($d_{0}$, $k$, $s$)}%
    [$\Theta(L)$]%
    [Define una palabra en el diccionario.]

    \InterfazFuncion{Obtener}{\In{k}{string}, \In{d}{diccStr($\alpha$)}}{$\alpha$}%
    [def?($d$, $k$)]%
    {$res \igobs$ obtener($k$, $d$)}%
    [$\Theta(L)$]%
    [Devuelve el significado asociado a la clave $k$.]

    \InterfazFuncion{Borrar}{\In{k}{string}}{diccStr($\alpha$), \Inout{d}{diccStr($\alpha$)}}%
    [$d \igobs d_{0}$ $\land$ def?($d$, $k$)]%
    {$d \igobs$ borrar($d_{0}$, $k$, $s$}%
    [$\Theta(L)$]%
    [Borra una palabra del diccionario.]

\end{Interfaz}
